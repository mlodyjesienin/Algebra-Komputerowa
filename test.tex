\documentclass{beamer}

\usepackage{xcolor} % Ensure color support
\usepackage{amsthm}
\makeatletter
% Define the block template
\defbeamertemplate{block definition begin}{rounded}[1][shadow=false]
{%
    \par\vskip\medskipamount%
    \begin{beamerboxesrounded}[upper=block title definition,lower=block body definition,#1]
    {\raggedright\usebeamerfont*{block title definition}\insertblocktitle}%
    \raggedright%
    \usebeamerfont{block body definition}%
}%

\defbeamertemplate{block definition end}{rounded}
{\end{beamerboxesrounded}}

% Define the custom definition block environment
\newenvironment<>{definitionblock}[1]{%
  \begin{actionenv}#2%
    \def\insertblocktitle{#1}%  % Store block title
    \par%
    \usebeamertemplate{block definition begin}%
}{%
  \par%
  \usebeamertemplate{block definition end}%
  \end{actionenv}%
}

% Associate definition theorem with the new environment
\def\th@definition{%
  \normalfont % body font
  \def\inserttheoremblockenv{definitionblock} % Use definitionblock
}
\makeatother

% Define custom colors
\definecolor{washed_red}{RGB}{230,143,143}
\setbeamercolor{block title definition}{bg=washed_red, fg=black}
\setbeamercolor{block body definition}{bg=white, fg=black} % Make body bg different to see change

% Sample content
\begin{document}

\begin{frame}
    \frametitle{Custom Definition Block Example}
    \begin{definition}{Custom Definition}
        This is an example of a custom definition block.
    \end{definition}
\end{frame}

\end{document}
